FULL DATABASE
🔹 1. USERS
CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100),
    email VARCHAR(150) UNIQUE,
    password VARCHAR(255),
    phone VARCHAR(20),
    avatar VARCHAR(255),

    role ENUM('user','owner','admin') DEFAULT 'user',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🔹 2. WALLETS (ví tiền)
CREATE TABLE wallets (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT UNIQUE,
    balance BIGINT DEFAULT 0,

    FOREIGN KEY (user_id) REFERENCES users(id)
);
🔹 3. DISTRICTS
CREATE TABLE districts (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255)
);
🔹 4. CATEGORIES
CREATE TABLE categories (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255)
);
🔹 5. MOTELS (phòng trọ)
CREATE TABLE motels (
    id INT AUTO_INCREMENT PRIMARY KEY,
    title VARCHAR(255),
    description TEXT,

    price INT,
    area INT,

    address VARCHAR(255),
    lat DECIMAL(10,8),
    lng DECIMAL(11,8),

    user_id INT,
    category_id INT,
    district_id INT,

    phone VARCHAR(20),

    count_view INT DEFAULT 0,
    status ENUM('pending','approved','hidden') DEFAULT 'pending',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (category_id) REFERENCES categories(id),
    FOREIGN KEY (district_id) REFERENCES districts(id)
);
🔹 6. MOTEL_IMAGES
CREATE TABLE motel_images (
    id INT AUTO_INCREMENT PRIMARY KEY,
    motel_id INT,
    image_url VARCHAR(255),

    FOREIGN KEY (motel_id) REFERENCES motels(id)
);
🔹 7. UTILITIES
CREATE TABLE utilities (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255)
);
🔹 8. MOTEL_UTILITIES (pivot)
CREATE TABLE motel_utilities (
    motel_id INT,
    utility_id INT,

    PRIMARY KEY (motel_id, utility_id),

    FOREIGN KEY (motel_id) REFERENCES motels(id),
    FOREIGN KEY (utility_id) REFERENCES utilities(id)
);
🔹 9. BOOKINGS (đặt cọc)
CREATE TABLE bookings (
    id INT AUTO_INCREMENT PRIMARY KEY,

    user_id INT,      -- người thuê
    motel_id INT,

    deposit_amount INT,
    checkin_date DATE,

    status ENUM(
        'pending',
        'paid',
        'accepted',
        'completed',
        'rejected',
        'cancelled'
    ) DEFAULT 'pending',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (motel_id) REFERENCES motels(id)
);
🔹 10. PAYMENTS (giữ tiền - escrow)
CREATE TABLE payments (
    id INT AUTO_INCREMENT PRIMARY KEY,

    booking_id INT,
    amount INT,

    method VARCHAR(50), -- vnpay, momo

    transaction_code VARCHAR(255),

    status ENUM(
        'pending',
        'held',
        'released',
        'refunded'
    ) DEFAULT 'pending',

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (booking_id) REFERENCES bookings(id)
);
🔹 11. TRANSACTIONS (log tiền)
CREATE TABLE transactions (
    id INT AUTO_INCREMENT PRIMARY KEY,

    from_user INT,
    to_user INT,

    amount INT,
    fee INT DEFAULT 0,

    type ENUM(
        'deposit',
        'release',
        'refund',
        'withdraw'
    ),

    booking_id INT,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
🔹 12. WITHDRAW REQUEST (rút tiền)
CREATE TABLE withdraw_requests (
    id INT AUTO_INCREMENT PRIMARY KEY,

    user_id INT,
    amount INT,

    status ENUM('pending','approved','rejected'),

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES users(id)
);
🔹 13. FAVORITES (yêu thích)
CREATE TABLE favorites (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT,
    motel_id INT,

    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (motel_id) REFERENCES motels(id)
);
🔹 14. REVIEWS (đánh giá)
CREATE TABLE reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,

    user_id INT,
    motel_id INT,

    rating INT CHECK (rating BETWEEN 1 AND 5),
    comment TEXT,

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (motel_id) REFERENCES motels(id)
);
🔥 15. VIP PACKAGES (kiếm tiền)
CREATE TABLE vip_packages (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(255),
    price INT,
    duration INT
);
🔹 16. USER_VIP
CREATE TABLE user_vip (
    id INT AUTO_INCREMENT PRIMARY KEY,

    user_id INT,
    package_id INT,

    start_date DATE,
    end_date DATE,

    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (package_id) REFERENCES vip_packages(id)
);
🔗 Quan hệ chính
users → motels (1-n)
users → bookings (1-n)
bookings → payments (1-1)
users → wallets (1-1)
wallets → transactions
motels → images (1-n)
motels → utilities (n-n)
